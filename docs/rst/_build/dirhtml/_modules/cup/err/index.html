[docs]class BaseCupException(Exception):
    """
    base cup Exception. All other cup Exceptions will inherit this.
    """
    def __init__(self, msg):
        self._msg = 'Cup module Exception:' + str(msg)

    def __str__(self):
        return repr(self._msg)
